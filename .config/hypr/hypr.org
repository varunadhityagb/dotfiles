* Shaders
** Dark Room Mode
#+begin_src glsl :tangle shaders/darkRoom.frag :mkdirp yes
#version 300 es

precision mediump float;
in vec2 v_texcoord;
layout(location = 0) out vec4 fragColor;
uniform sampler2D tex;

void main() {

    vec4 pixColor = texture(tex, v_texcoord);

    float brightness = -0.05;
    float contrast = 1.1;

    pixColor.rgb = (pixColor.rgb - 0.5) * contrast + 0.5;
    pixColor.rgb += brightness;

    fragColor = pixColor;
}
#+end_src
** Focus Mode
#+begin_src glsl :tangle shaders/focusMode.frag :mkdirp yes
#version 300 es

precision mediump float;
in vec2 v_texcoord;
layout(location = 0) out vec4 fragColor;
uniform sampler2D tex;

void main() {

    vec4 pixColor = texture(tex, v_texcoord);

    float gray = dot(pixColor.rgb, vec3(0.299, 0.587, 0.114));
    pixColor.rgb = vec3(gray);

    fragColor = pixColor;
}
#+end_src
** Night Mode
#+begin_src glsl :tangle shaders/nightMode.frag :mkdirp yes
#version 300 es

precision mediump float;
in vec2 v_texcoord;
layout(location = 0) out vec4 fragColor;
uniform sampler2D tex;

void main() {

    vec4 pixColor = texture(tex, v_texcoord);

    pixColor.r *= 1.05;
    pixColor.g *= 0.98;
    pixColor.b *= 0.85;

    fragColor = pixColor;
}
#+end_src
** Sunlight Mode
#+begin_src glsl :tangle shaders/sunlightMode.frag :mkdirp yes
#version 300 es
precision mediump float;

in vec2 v_texcoord;
out vec4 fragColor;

uniform sampler2D tex;

void main() {
    vec4 color = texture(tex, v_texcoord);

    // Contrast boost
    float contrast = 1.9;
    color.rgb = (color.rgb - 0.5) * contrast + 0.5;

    // Slight brightness lift
    color.rgb += 0.15;

    // Mild gamma correction
    color.rgb = pow(color.rgb, vec3(1.0 / 1.08));

    fragColor = color;
}
